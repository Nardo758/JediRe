const fonts = "'Instrument Sans', 'DM Sans', -apple-system, sans-serif";
const mono = "'JetBrains Mono', 'Fira Code', monospace";

const colors = {
  bg: "#0B0F1A",
  card: "rgba(255,255,255,0.025)",
  cardBorder: "rgba(255,255,255,0.06)",
  text: "#E2E8F0",
  muted: "#64748B",
  dim: "#334155",
  accent: "#3B82F6",
  green: "#22C55E",
  yellow: "#F59E0B",
  red: "#EF4444",
  purple: "#8B5CF6",
  cyan: "#06B6D4",
  pink: "#EC4899",
  orange: "#F97316",
};

function Pill({ children, color = colors.muted }) {
  return (
    <span style={{
      fontSize: 9, fontWeight: 700, padding: "2px 8px", borderRadius: 5,
      color, background: `${color}14`, border: `1px solid ${color}22`,
      letterSpacing: "0.05em", whiteSpace: "nowrap",
    }}>{children}</span>
  );
}

function SourceBadge({ type }) {
  const map = {
    S: { label: "STRUCTURED", color: colors.accent, icon: "📊" },
    N: { label: "NEWS", color: colors.orange, icon: "📰" },
    D: { label: "DOCUMENT", color: colors.purple, icon: "📄" },
  };
  const s = map[type] || map.S;
  return <Pill color={s.color}>{s.icon} {s.label}</Pill>;
}

function VerificationBadge({ status }) {
  const map = {
    verified: { label: "✅ VERIFIED", color: colors.green },
    partial: { label: "⚠️ PARTIAL", color: colors.yellow },
    contradicted: { label: "❌ CONTRADICTED", color: colors.red },
    unverifiable: { label: "❓ UNVERIFIABLE", color: colors.muted },
  };
  const s = map[status] || map.unverifiable;
  return <Pill color={s.color}>{s.label}</Pill>;
}

function SectionHeader({ icon, title, subtitle, badge, isNew }) {
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 10, marginBottom: 12 }}>
      <span style={{ fontSize: 18 }}>{icon}</span>
      <div>
        <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
          <span style={{ fontSize: 15, fontWeight: 700, color: "#F8FAFC" }}>{title}</span>
          {isNew && <Pill color={colors.green}>NEW</Pill>}
          {badge && <Pill color={colors.accent}>{badge}</Pill>}
        </div>
        {subtitle && <div style={{ fontSize: 11, color: colors.muted, marginTop: 2 }}>{subtitle}</div>}
      </div>
    </div>
  );
}

function Card({ children, style = {}, highlight }) {
  return (
    <div style={{
      background: colors.card, border: `1px solid ${highlight || colors.cardBorder}`,
      borderRadius: 12, padding: 18, ...style,
    }}>{children}</div>
  );
