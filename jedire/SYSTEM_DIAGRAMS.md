### Module Loading Logic

```typescript
// Frontend: Module gating based on subscription
const useModuleAccess = (dealId: string, moduleName: string) => {
  const { user } = useAuth();
  const { selectedDeal } = useDealStore();
  
  const tier = user.subscription.tier; // 'basic' | 'pro' | 'enterprise'
  
  const MODULE_ACCESS = {
    basic: ['map', 'properties', 'pipeline_basic'],
    pro: ['map', 'properties', 'pipeline_advanced', 'strategy', 'market'],
    enterprise: ['map', 'properties', 'pipeline_advanced', 'strategy', 'market', 'reports', 'team']
  };
  
  const hasAccess = MODULE_ACCESS[tier].includes(moduleName);
  
  return {
    hasAccess,
    upgradeRequired: !hasAccess,
    nextTier: tier === 'basic' ? 'pro' : 'enterprise'
  };
};
```

### PostGIS Queries

```sql
-- Check if property is inside deal boundary
SELECT p.id, p.address, p.rent
FROM properties p
JOIN deals d ON ST_Contains(d.boundary, ST_Point(p.lng, p.lat))
WHERE d.id = 'deal-uuid';

-- Calculate area in acres
SELECT 
  ST_Area(boundary::geography) / 4046.86 AS acres
FROM deals
WHERE id = 'deal-uuid';

-- Find deals near a point (10 mile radius)
SELECT id, name, 
  ST_Distance(boundary::geography, ST_Point(-84.388, 33.749)::geography) / 1609.34 AS distance_miles
FROM deals
WHERE ST_DWithin(boundary::geography, ST_Point(-84.388, 33.749)::geography, 16093.4) -- 10 miles in meters
ORDER BY distance_miles;
```
