## Implementation Notes

### Database Schema Updates Needed

```sql
-- Deals table
CREATE TABLE deals (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  name VARCHAR(255),
  boundary GEOMETRY(POLYGON, 4326), -- Map boundary
  project_type VARCHAR(50), -- 'multifamily', 'mixed_use', etc.
  project_intent TEXT,
  target_units INT,
  budget NUMERIC,
  timeline_start DATE,
  timeline_end DATE,
  subscription_tier VARCHAR(20), -- 'basic', 'pro', 'enterprise'
  created_at TIMESTAMP,
  updated_at TIMESTAMP
);

-- Deal modules (tracks enabled features per deal)
CREATE TABLE deal_modules (
  id UUID PRIMARY KEY,
  deal_id UUID REFERENCES deals(id),
  module_name VARCHAR(50), -- 'map', 'properties', 'strategy', etc.
  is_enabled BOOLEAN,
  settings JSONB -- Module-specific settings
);

-- Deal annotations (map markers, notes, etc.)
CREATE TABLE deal_annotations (
  id UUID PRIMARY KEY,
  deal_id UUID REFERENCES deals(id),
  type VARCHAR(50), -- 'marker', 'polygon', 'text'
  geometry GEOMETRY,
  content TEXT,
  created_at TIMESTAMP
);

-- Link properties to deals
CREATE TABLE deal_properties (
  deal_id UUID REFERENCES deals(id),
  property_id UUID REFERENCES properties(id),
  relationship VARCHAR(50), -- 'comparable', 'target', 'competitor'
  notes TEXT,
  PRIMARY KEY (deal_id, property_id)
);

-- Link emails to deals (AI-powered)
CREATE TABLE deal_emails (
  deal_id UUID REFERENCES deals(id),
  email_id UUID REFERENCES emails(id),
  confidence_score DECIMAL(3,2), -- 0.00 to 1.00
  linked_by VARCHAR(50), -- 'ai' or 'manual'
  PRIMARY KEY (deal_id, email_id)
);
```
