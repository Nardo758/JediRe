function parseAddress(address: string): { city?: string; state?: string } {
  const parts = address.split(',').map(p => p.trim());
  
  if (parts.length >= 2) {
    const stateZip = parts[parts.length - 1];
    const city = parts[parts.length - 2];
    const state = stateZip.split(' ')[0];
    
    return { city, state };
  }
  
  return {};
}

/**
 * Determine county from city/state
 */
function getCounty(city?: string, state?: string): string | null {
  if (!city || !state) return null;
  
  const cityLower = city.toLowerCase();
  const stateUpper = state.toUpperCase();
  
  if (stateUpper === 'GA' || stateUpper === 'GEORGIA') {
    if (cityLower.includes('atlanta') || cityLower.includes('buckhead') || 
        cityLower.includes('midtown') || cityLower.includes('inman') ||
        cityLower.includes('westside') || cityLower.includes('east atlanta') ||
        cityLower.includes('grant park') || cityLower.includes('reynoldstown') ||
        cityLower.includes('college park')) {
      return 'Fulton County, GA';
    }
    if (cityLower.includes('decatur')) {
      return 'DeKalb County, GA';
    }
    if (cityLower.includes('marietta')) {
      return 'Cobb County, GA';
    }
    if (cityLower.includes('sandy springs') || cityLower.includes('alpharetta')) {
      return 'Fulton County, GA';
    }
  }
  
  if (stateUpper === 'FL' || stateUpper === 'FLORIDA') {
    if (cityLower.includes('miami')) {
      return 'Miami-Dade County, FL';
    }
    if (cityLower.includes('west palm beach') || cityLower.includes('palm beach')) {
      return 'Palm Beach County, FL';
    }
  }
  
  return null;
}

function cleanAddressForQuery(address: string): string {
  // Take just the street address (first part before comma)

    .trim();
