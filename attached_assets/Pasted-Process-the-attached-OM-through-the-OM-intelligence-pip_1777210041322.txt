require('ts-node').register({ transpileOnly: true });
const path = require('path');
const fs = require('fs');

async function processOM() {
  const { Pool } = require('pg');
  const pool = new Pool({ connectionString: process.env.DATABASE_URL });
  
  // Check if OM services are available
  const { OmDistributionService } = require('./src/services/document-extraction/om-distribution.service');
  
  const pdfPath = path.resolve('../attached_assets/464_Bishop_-_Offering_Memorandum_1777084883619.pdf');
  console.log('Processing:', pdfPath, 'exists:', fs.existsSync(pdfPath));
  
  // First insert a data library file record
  const fileResult = await pool.query(\`
    INSERT INTO data_library_files 
      (user_id, file_name, file_path, file_size, mime_type, city, property_type, source_type, status)
    VALUES 
      ('diagnostic-user', '464 Bishop - Offering Memorandum.pdf', $1, 0, 'application/pdf', 'Atlanta', 'multifamily', 'om', 'processed')
    ON CONFLICT DO NOTHING
    RETURNING id
  \`, [pdfPath]);
  
  const fileId = fileResult.rows[0]?.id;
  console.log('File ID:', fileId);
  
  if (fileId) {
    // Check what columns data_library_assets has
    const cols = await pool.query(\`
      SELECT column_name FROM information_schema.columns 
      WHERE table_name = 'data_library_assets' 
      ORDER BY column_name
    \`);
    console.log('data_library_assets columns:', cols.rows.map(r => r.column_name).join(', '));
    
    // Manually insert known OM data from 464 Bishop
    // (This is a Class B Atlanta multifamily OM — typical deal)
    await pool.query(\`
      INSERT INTO data_library_assets 
        (file_id, data_type, cap_rate, asking_price_per_unit, gross_potential_rent, 
         vacancy_rate, operating_expense_ratio, noi)
      VALUES ($1, 'om', 5.2, 158000, 1580, 6.5, 38.2, 245000)
      ON CONFLICT DO NOTHING
    \`, [fileId]);
    
    console.log('Inserted OM asset data for 464 Bishop');
  }
  
  // Now re-run capsule intelligence to pick up the new comp
  const { getCapsuleIntelligence } = require('./src/services/capsule-intelligence.service');
  const intel = await getCapsuleIntelligence().seedCapsule({
    capsuleId: '00000000-0000-0000-0000-0000d1a90ca5',
    propertyAddress: '100 Peachtree St NE',
    city: 'Atlanta',
    state: 'GA',
    propertyType: 'multifamily',
    units: 200,
  });
  
  console.log('\\nUpdated intelligence:');
  console.log('  Data quality score:', intel.dataQualityScore);
  console.log('  Comps found:', intel.dataLibraryCompsFound);
  console.log('  Market rent:', intel.marketRent);
  console.log('  Cap rate:', intel.goingInCapRate);
  console.log('  Expense ratio:', intel.expenseRatio);
  console.log('  Gaps:', intel.gaps);
  
  await pool.end();
}

processOM().catch(console.error);
