> RocketManOpenClaw2:
In Replit's Shell tab, run:

cd backend && npx ts-node -e "
const { Pool } = require('pg');
const fs = require('fs');
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

async function run() {
  const migrations = [
    'src/database/migrations/20260422_agent_system_tables.sql',
    'src/database/migrations/20260422_morning_briefs_table.sql',
    'src/database/migrations/20260422_discovery_tables.sql'
  ];
  
  for (const file of migrations) {
    console.log('Running:', file);
    const sql = fs.readFileSync(file, 'utf8');
    await pool.query(sql);
    console.log('✓ Done');
  }
  
  await pool.end();
  console.log('All migrations complete!');
}

run().catch(e => { console.error(e); process.exit(1); });
"

Or if you prefer to run them one at a time:

# Pull latest code first
git pull origin master

# Then run each migration
psql $DATABASE_URL -f src/database/migrations/20260422_agent_system_tables.sql
psql $DATABASE_URL -f src/database/migrations/20260422_morning_briefs_table.sql
psql $DATABASE_URL -f src/database/migrations/20260422_discovery_tables.sql

If psql isn't available, use the Replit Database tool or the ts-node command above.
