                              <span key={ii} style={{ fontSize: 8, color: T.text.secondary, background: T.bg.header, padding: "2px 6px" }}>{item}</span>
                            ))}
                          </div>
                        )}
                        {c.note && <div style={{ fontSize: 8, color: T.text.muted, marginTop: 4, fontFamily: T.font.label, lineHeight: 1.4 }}>{c.note}</div>}
                      </div>
                    </div>
                  ))}

                  {/* Wiring info */}
                  <div style={{ marginTop: 16, display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12 }}>
                    <div style={{ padding: "8px 12px", background: T.bg.panel, border: `1px solid ${T.border.subtle}` }}>
                      <div style={{ fontSize: 8, color: T.text.muted, letterSpacing: 1, marginBottom: 4 }}>DATA SOURCE</div>
                      <div style={{ fontSize: 9, color: T.text.green, fontFamily: T.font.label }}>{selectedMarket.dataSource}</div>
                    </div>
                    <div style={{ padding: "8px 12px", background: T.bg.panel, border: `1px solid ${T.border.subtle}` }}>
                      <div style={{ fontSize: 8, color: T.text.muted, letterSpacing: 1, marginBottom: 4 }}>EXISTING CODE</div>
                      <div style={{ fontSize: 9, color: T.text.orange, fontFamily: T.font.label }}>{selectedMarket.existingCode}</div>
                    </div>
                  </div>
                  <div style={{ marginTop: 8, padding: "8px 12px", background: T.bg.header, border: `1px solid ${T.text.amber}33` }}>
                    <div style={{ fontSize: 8, color: T.text.amber, fontWeight: 700, letterSpacing: 1, marginBottom: 2 }}>WIRE ACTION</div>
                    <div style={{ fontSize: 9, color: T.text.primary, fontFamily: T.font.label, lineHeight: 1.4 }}>{selectedMarket.wireAction}</div>
                  </div>
                </div>
              )}
            </div>
          </>
        )}

        {/* ═══ DEAL CAPSULE VIEW ═══ */}
        {view === "capsule" && (
          <>
            {/* LEFT: Tab list with status indicators */}
            <div style={{ width: 240, borderRight: `1px solid ${T.border.subtle}`, overflow: "auto", flexShrink: 0 }}>
              <div style={{ padding: "10px 12px 6px", fontSize: 8, color: T.text.muted, letterSpacing: 1 }}>DEAL CONTEXT → F1–F12</div>
              
              {/* Summary bar */}
              <div style={{ margin: "4px 8px 8px", display: "flex", gap: 4 }}>
                {Object.entries(statusColors).map(([k, c]) => {
                  const count = DEAL_TABS.filter(t => t.uiStatus === k).length;
                  return (
                    <div key={k} style={{ flex: 1, padding: "4px", background: c + "12", border: `1px solid ${c}33`, textAlign: "center" }}>
                      <div style={{ fontSize: 14, fontWeight: 800, color: c }}>{count}</div>
                      <div style={{ fontSize: 6, color: c, letterSpacing: 0.5 }}>{k.toUpperCase()}</div>
                    </div>
                  );
                })}
              </div>

              {DEAL_TABS.map(tab => {
                const sc = statusColors[tab.uiStatus];
                return (
                  <button key={tab.fkey} onClick={() => setSelectedDealTab(tab.fkey)} style={{
                    display: "flex", alignItems: "center", justifyContent: "space-between", width: "100%",
                    textAlign: "left", padding: "6px 10px", border: "none", cursor: "pointer",
                    background: selectedDealTab === tab.fkey ? T.bg.active : "transparent",
                    borderLeft: selectedDealTab === tab.fkey ? `3px solid ${T.text.amber}` : "3px solid transparent",
                    borderBottom: `1px solid ${T.border.subtle}`,
                  }}>
                    <div>
                      <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
                        <span style={{ fontSize: 8, color: T.text.muted }}>{tab.fkey}</span>
                        <span style={{ fontSize: 10, fontWeight: 600, color: selectedDealTab === tab.fkey ? T.text.amber : T.text.primary, fontFamily: T.font.mono }}>{tab.label}</span>
                      </div>
                      <div style={{ fontSize: 7, color: T.text.muted, marginTop: 1 }}>{tab.module}</div>
                    </div>
                    <div style={{ display: "flex", alignItems: "center", gap: 6 }}>
                      {tab.actions.length > 0 && <span style={{ fontSize: 8, color: T.text.amber, fontWeight: 700 }}>{tab.actions.length}</span>}
                      <span style={{ width: 6, height: 6, borderRadius: "50%", background: sc }} />
                    </div>
                  </button>
                );
              })}

              <div style={{ margin: 8, padding: "6px 8px", background: T.text.red + "0A", border: `1px solid ${T.text.red}22`, borderRadius: 2 }}>
                <div style={{ fontSize: 8, color: T.text.red, fontWeight: 700, letterSpacing: 1, marginBottom: 2 }}>WORST OFFENDERS</div>
                <div style={{ fontSize: 9, color: T.text.secondary, fontFamily: T.font.label, lineHeight: 1.5 }}>
                  <span style={{ color: T.text.red, fontWeight: 600 }}>F2 Zoning</span> and <span style={{ color: T.text.red, fontWeight: 600 }}>F9 Comps</span> — 
                  both still fully white Tailwind UI. These need surgical token replacement.
                </div>
              </div>
            </div>

            {/* RIGHT: Tab detail */}
            <div style={{ flex: 1, overflow: "auto", animation: "fadeIn 0.15s" }}>
              {selectedDeal && (
                <div style={{ padding: 16 }}>
                  {/* Header */}
                  <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", padding: "8px 12px", background: T.bg.header, borderTop: `3px solid ${statusColors[selectedDeal.uiStatus]}`, borderBottom: `1px solid ${T.border.subtle}`, marginBottom: 16 }}>
                    <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
                      <span style={{ fontSize: 12, fontWeight: 800, color: T.text.white }}>{selectedDeal.fkey} {selectedDeal.label}</span>
                      <span style={{ fontSize: 9, color: T.text.muted }}>{selectedDeal.module}</span>
                    </div>
                    <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
                      <span style={{
                        fontSize: 8, fontWeight: 700, letterSpacing: 0.5, padding: "2px 8px",
                        color: statusColors[selectedDeal.uiStatus],
                        background: statusColors[selectedDeal.uiStatus] + "18",
                        border: `1px solid ${statusColors[selectedDeal.uiStatus]}33`,
                      }}>{statusLabels[selectedDeal.uiStatus]}</span>
                      <div style={{ display: "flex", alignItems: "center", gap: 4 }}>
                        <div style={{ width: 60, height: 4, background: T.bg.terminal, borderRadius: 2, overflow: "hidden" }}>
                          <div style={{ width: `${selectedDeal.confidence}%`, height: "100%", background: selectedDeal.confidence >= 80 ? T.text.green : selectedDeal.confidence >= 50 ? T.text.amber : T.text.red, borderRadius: 2 }} />
                        </div>
                        <span style={{ fontSize: 8, fontWeight: 700, color: T.text.secondary }}>{selectedDeal.confidence}%</span>
                      </div>
                    </div>
                  </div>

                  {/* Status notes */}
                  <div style={{ padding: "8px 12px", background: T.bg.panel, borderLeft: `3px solid ${statusColors[selectedDeal.uiStatus]}`, marginBottom: 16 }}>
                    <div style={{ fontSize: 8, color: T.text.muted, letterSpacing: 1, marginBottom: 4 }}>STATUS ASSESSMENT</div>
                    <div style={{ fontSize: 10, color: T.text.primary, fontFamily: T.font.label, lineHeight: 1.5 }}>{selectedDeal.notes}</div>
                  </div>

                  {/* Recommendation (if present) */}
                  {selectedDeal.recommendation && (
                    <div style={{ padding: "8px 12px", background: T.text.amber + "08", border: `1px solid ${T.text.amber}33`, marginBottom: 16 }}>
                      <div style={{ fontSize: 8, color: T.text.amber, fontWeight: 700, letterSpacing: 1, marginBottom: 4 }}>RECOMMENDATION</div>
                      <div style={{ fontSize: 10, color: T.text.primary, fontFamily: T.font.label, lineHeight: 1.5 }}>{selectedDeal.recommendation}</div>
                    </div>
                  )}

                  {/* Actions */}
                  {selectedDeal.actions.length > 0 ? (
                    <>
                      <div style={{ fontSize: 9, color: T.text.secondary, fontWeight: 700, letterSpacing: 1, marginBottom: 8 }}>
                        CONVERSION ACTIONS ({selectedDeal.actions.length})
                      </div>
                      {selectedDeal.actions.map((action, i) => (
                        <div key={i} style={{
                          display: "grid", gridTemplateColumns: "44px 44px 1fr", gap: 8, alignItems: "center",
                          padding: "6px 10px", background: i % 2 === 0 ? T.bg.panel : T.bg.panelAlt,
                          borderBottom: `1px solid ${T.border.subtle}`, borderLeft: action.priority === "P0" ? `3px solid ${T.text.red}` : `3px solid transparent`,
                        }}>
                          <span style={{
                            fontSize: 8, fontWeight: 700, textAlign: "center", padding: "2px 0",
                            color: action.priority === "P0" ? T.text.red : action.priority === "P1" ? T.text.amber : T.text.muted,
                          }}>{action.priority}</span>
                          <span style={{
                            fontSize: 8, fontWeight: 700, textAlign: "center", padding: "2px 4px",
                            color: effortColors[action.effort], background: effortColors[action.effort] + "18",
                          }}>{action.effort}</span>
                          <span style={{ fontSize: 9, color: T.text.primary, fontFamily: T.font.label }}>{action.task}</span>
                        </div>
                      ))}
                    </>
                  ) : (
                    <div style={{ padding: "12px", background: T.text.green + "08", border: `1px solid ${T.text.green}22`, textAlign: "center" }}>
                      <span style={{ fontSize: 10, color: T.text.green, fontWeight: 600 }}>No conversion actions needed — this tab is Bloomberg-native.</span>
                    </div>
                  )}

                  {/* Token replacement cheatsheet for white/partial tabs */}
                  {(selectedDeal.uiStatus === "white" || selectedDeal.uiStatus === "partial") && (
                    <div style={{ marginTop: 20, padding: "10px 12px", background: T.bg.header, border: `1px solid ${T.border.medium}` }}>
                      <div style={{ fontSize: 9, color: T.text.cyan, fontWeight: 700, letterSpacing: 1, marginBottom: 8 }}>TOKEN REPLACEMENT CHEATSHEET</div>
                      <div style={{ display: "grid", gridTemplateColumns: "1fr 20px 1fr", gap: "4px 8px", fontSize: 9 }}>
                        {[
                          ["bg-white", "→", "T.bg.panel (#0F1319)"],
                          ["bg-gray-50", "→", "T.bg.panelAlt (#131821)"],
                          ["bg-gray-100", "→", "T.bg.header (#1A1F2E)"],
                          ["text-gray-900", "→", "T.text.primary (#E8ECF1)"],
                          ["text-gray-600", "→", "T.text.secondary (#8B95A5)"],
                          ["text-gray-400", "→", "T.text.muted (#4A5568)"],
                          ["text-blue-600", "→", "T.text.amber (#F5A623)"],
                          ["border-gray-200", "→", "T.border.subtle (#1E2538)"],
                          ["border-gray-300", "→", "T.border.medium (#2A3348)"],
                          ["shadow-md", "→", "REMOVE (no shadows in terminal)"],
                          ["rounded-lg", "→", "borderRadius: 2 (max)"],
                          ["hover:bg-gray-50", "→", "hover T.bg.hover (#1E2538)"],
                        ].map((row, i) => (
                          <React.Fragment key={i}>
                            <span style={{ color: T.text.red, fontFamily: T.font.mono }}>{row[0]}</span>
                            <span style={{ color: T.text.muted, textAlign: "center" }}>{row[1]}</span>
                            <span style={{ color: T.text.green, fontFamily: T.font.mono }}>{row[2]}</span>
                          </React.Fragment>
                        ))}
                      </div>
                    </div>
                  )}
                </div>
              )}
            </div>
          </>
        )}
      </div>
    </div>
  );
}
