## Architecture Overview

### Core Concept
**Modular Plugin System** - Users purchase and configure modules to build custom workflows

**Key Principles:**
1. **Purchasable Modules** - Marketplace model, not just toggles
2. **Custom Strategies** - Users create their own analysis strategies
3. **User-Controlled Layout** - Drag-to-reorder module tabs
4. **Setup Wizard** - Guided module selection during onboarding
5. **Per-Deal Activation** - Enable different modules per deal

## Database Schema

### New Tables

**1. `modules` - Available modules in marketplace**

```sql
CREATE TABLE modules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  slug VARCHAR(50) UNIQUE NOT NULL,
  category VARCHAR(50) NOT NULL, -- 'financial', 'development', 'diligence', 'market', 'collaboration', 'portfolio'
  description TEXT,
  long_description TEXT,
  icon VARCHAR(50),
  price_monthly NUMERIC(10,2),
  price_annual NUMERIC(10,2),
  is_free BOOLEAN DEFAULT false,
  tier_required VARCHAR(20), -- 'basic', 'pro', 'enterprise'
  powered_by_agent VARCHAR(50), -- maps to AI agent
  features JSONB, -- array of feature bullets
  screenshots JSONB, -- array of screenshot URLs
  install_count INTEGER DEFAULT 0,
  rating_average NUMERIC(3,2),
  rating_count INTEGER DEFAULT 0,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_modules_category ON modules(category);
CREATE INDEX idx_modules_price ON modules(price_monthly);
CREATE INDEX idx_modules_free ON modules(is_free);
```

---

**2. `user_modules` - Modules purchased/owned by user**

```sql
CREATE TABLE user_modules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  module_id UUID REFERENCES modules(id) ON DELETE CASCADE,
  is_enabled BOOLEAN DEFAULT true,
  global_position INTEGER, -- order in global module list
  configuration JSONB, -- module settings
  subscription_status VARCHAR(20), -- 'trial', 'active', 'cancelled', 'expired'
  trial_ends_at TIMESTAMP,
  subscription_ends_at TIMESTAMP,
  installed_at TIMESTAMP DEFAULT NOW(),
  last_used_at TIMESTAMP,
  
  UNIQUE(user_id, module_id)
);

CREATE INDEX idx_user_modules_user ON user_modules(user_id);
CREATE INDEX idx_user_modules_enabled ON user_modules(is_enabled);
```

---

**3. `deal_modules` - Modules enabled per deal**

```sql
CREATE TABLE deal_modules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  deal_id UUID REFERENCES deals(id) ON DELETE CASCADE,
  module_id UUID REFERENCES modules(id) ON DELETE CASCADE,
  is_enabled BOOLEAN DEFAULT true,
  position INTEGER, -- tab order for this deal
  configuration JSONB, -- deal-specific module settings
  enabled_at TIMESTAMP DEFAULT NOW(),
  last_accessed_at TIMESTAMP,
  
  UNIQUE(deal_id, module_id)
);

CREATE INDEX idx_deal_modules_deal ON deal_modules(deal_id);
CREATE INDEX idx_deal_modules_position ON deal_modules(deal_id, position);
```

---

**4. `custom_strategies` - User-created strategies**

```sql
CREATE TABLE custom_strategies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  name VARCHAR(100) NOT NULL,
  slug VARCHAR(50) NOT NULL,
  description TEXT,
  icon VARCHAR(50) DEFAULT 'ğŸ¯',
  is_active BOOLEAN DEFAULT true,
  is_default BOOLEAN DEFAULT false,
  is_public BOOLEAN DEFAULT false, -- shared with community
  
  -- Strategy parameters
  hold_period_min INTEGER,
  hold_period_target INTEGER,
  hold_period_max INTEGER,
  
  exit_strategies JSONB, -- ['sale', 'refinance', '1031', 'hold']
  
  revenue_assumptions JSONB,
  expense_assumptions JSONB,
  financing_assumptions JSONB,
  value_add_assumptions JSONB,
  target_returns JSONB,
  
  ai_scoring_rules JSONB,
  confidence_threshold INTEGER DEFAULT 70,
  
  use_count INTEGER DEFAULT 0,
  avg_confidence NUMERIC(5,2),
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  
  UNIQUE(user_id, slug)
);

CREATE INDEX idx_custom_strategies_user ON custom_strategies(user_id);
CREATE INDEX idx_custom_strategies_active ON custom_strategies(is_active);
CREATE INDEX idx_custom_strategies_public ON custom_strategies(is_public);
```
